<algorithm> ::= "ALGORITIMO " "\"" <name_convention> "\";" "\nVARIAVEIS\n" <var_declaration_section> "INICIO" <instructions_section> "\nFIM_ALGORITMO"

<name_convention> ::= (<char>)+ (<char> | "_" | <int_number>)*

<var_declaration_section> ::= "\n" | <var_declaration>+

<var_declaration> ::= "\t" <name_convention> (", " <name_convention>) " : " <var_type> "\n"

<var_type> ::= "Inteiro" | "Real" | "Texto" | "Logico"

<instructions_section> ::= "\n" | <instruction>*
<instruction> ::= <print>

<print> ::= "\nESCREVA(" ");"

<int_number> ::= [0-9]
<text> ::= (<char>+ | " "*)+
<char> ::= <lowercase_char> | <uppercase_char>
<lowercase_char> ::= [a-z]
<uppercase_char> ::= [A-Z]
